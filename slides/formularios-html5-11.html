<script type="text/javascript">
    function customEmailInvalid(input){
        //Si no es valido
        if(!input.validity.valid) {
            //Cambiar el mensaje de error a mostrar
            input.setCustomValidity("Algo falla");
        }
    }
</script>
<form>
    <div class="form-group">
        <label for="email1">Email</label>
        <input type="email" class="form-control" id="email1" placeholder="mail@domain.com" oninvalid="customEmailInvalid(this)" required="required">
    </div>
    <div class="form-group">
        <button type="submit" class="btn btn-default">Submit</button>
        <label for="checkbox"></label>
        <input type="checkbox" id="checkbox"> Accept conditions
    </div>
</form>
